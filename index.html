import pygame import random # 게임 초기화 pygame.init() # 화면 설정 WIDTH = 800 HEIGHT = 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Simple Shooting Game") # 색상 정의 WHITE = (255, 255, 255) RED = (255, 0, 0) BLUE = (0, 0, 255) # 플레이어 설정 player_width = 50 player_height = 50 player_x = WIDTH // 2 - player_width // 2 player_y = HEIGHT - player_height - 10 player_speed = 5 # 총알 설정 bullet_width = 5 bullet_height = 15 bullet_speed = 7 bullets = [] # 적 설정 enemy_width = 50 enemy_height = 50 enemy_speed = 3 enemies = [] # 점수 score = 0 font = pygame.font.Font(None, 36) # 게임 루프 running = True clock = pygame.time.Clock() while running: for event in pygame.event.get(): if event.type == pygame.QUIT: running = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE: bullets.append([player_x + player_width // 2 - bullet_width // 2, player_y]) # 플레이어 이동 keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player_x > 0: player_x -= player_speed if keys[pygame.K_RIGHT] and player_x < WIDTH - player_width: player_x += player_speed # 총알 이동 for bullet in bullets[:]: bullet[1] -= bullet_speed if bullet[1] < 0: bullets.remove(bullet) # 적 생성 if len(enemies) < 5: enemy_x = random.randint(0, WIDTH - enemy_width) enemies.append([enemy_x, -enemy_height]) # 적 이동 for enemy in enemies[:]: enemy[1] += enemy_speed if enemy[1] > HEIGHT: enemies.remove(enemy) # 충돌 감지 for enemy in enemies[:]: for bullet in bullets[:]: if (enemy[0] < bullet[0] < enemy[0] + enemy_width and enemy[1] < bullet[1] < enemy[1] + enemy_height): enemies.remove(enemy) bullets.remove(bullet) score += 1 # 화면 그리기 screen.fill((0, 0, 0)) # 플레이어 그리기 pygame.draw.rect(screen, BLUE, (player_x, player_y, player_width, player_height)) # 총알 그리기 for bullet in bullets: pygame.draw.rect(screen, WHITE, (bullet[0], bullet[1], bullet_width, bullet_height)) # 적 그리기 for enemy in enemies: pygame.draw.rect(screen, RED, (enemy[0], enemy[1], enemy_width, enemy_height)) # 점수 표시 score_text = font.render(f"Score: {score}", True, WHITE) screen.blit(score_text, (10, 10)) pygame.display.flip() clock.tick(60) pygame.quit()
